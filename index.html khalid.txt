<script>
    function checkQuiz() {
        let score = 0;
        let totalQuestions = 3;
        
        // التحقق من الإجابات
        let q1 = document.querySelector('input[name="q1"]:checked');
        let q2 = document.querySelector('input[name="q2"]:checked');
        let q3 = document.querySelector('input[name="q3"]:checked');

        if(!q1 || !q2 || !q3) {
            alert("يرجى الإجابة على جميع الأسئلة!");
            return;
        }

        if(q1.value === "true") score++;
        if(q2.value === "true") score++;
        if(q3.value === "true") score++;

        let resultDiv = document.getElementById('result');
        resultDiv.innerHTML = "درجتك هي: " + score + " من " + totalQuestions;
        
        if(score === totalQuestions) {
            resultDiv.style.color = "green";
            resultDiv.innerHTML += " - أحسنت يا خالد، إجاباتك مثالية!";
        } else {
            resultDiv.style.color = "red";
            resultDiv.innerHTML += " - حاول مرة أخرى لتحسين نتيجتك.";
        }
    }
</script>
